解法一
```go
/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func reverseList(head *ListNode) *ListNode {
    //定义双指针   一个pre  一个cur     
    var pre *ListNode
    cur :=head
    for cur!=nil {                //当cur移动到nil 时     pre 即到了head的末尾
        next := cur.Next
        cur.Next= pre             //将cur.Next 指向 pre 

        pre = cur                 //pre 往后移动
        cur = next                //cur 往后移动
    }
    return pre
}
```
